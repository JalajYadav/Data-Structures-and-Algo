int findNumberOfTriangles(int arr[], int n)
    {
        // code here
        sort(arr, arr + n);
        int count = 0;
        for(int i =0;i<=n-3;i++){
            
            int k = i+2; 
            for(int j =i+1;j<=n-2;j++){
                while(k<n and (arr[i]+arr[j]>arr[k])){
                    k++;
                }
            if(k>j)
                count += k - j - 1;
            }
        }
        return count;
    }
